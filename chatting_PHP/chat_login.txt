<?php
if(isset($_POST["u_id"]) && isset($_POST["u_pw"])){                        // POST로 값이 들어올 경우

        $user_ID = $_POST["u_id"];
        $user_PW = $_POST["u_pw"];

        include_once 'Config.php';
        $conn = mysqli_connect(DB_HOST, DB_USER, DB_PW, DB_NAME); 
        $sql = "SELECT * FROM users WHERE userEmail = '$user_ID' and userPW = '$user_PW'";       // 사용자 이메일과 비밀번호가 DB 내 존재 여부 확인,
        $result = mysqli_query($conn, $sql); 
        if($result){
                $count = mysqli_num_rows($result);                            // SELECT 문에 해당하는 Query가 존재 할 시,
                if($count > 0){
                        $data_row = mysqli_fetch_array($result);
                        echo "$data_row[0], $data_row[1], $data_row[2]";      // 사용자 정보 반환
                }
                else
                        echo "Fail";
        }else
                echo "Fail";

        mysqli_close($conn);
}
?>
