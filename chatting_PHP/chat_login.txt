<?php
if(isset($_POST["u_id"]) && isset($_POST["u_pw"])){                        // Input type -> POST 

        $user_ID = $_POST["u_id"];
        $user_PW = $_POST["u_pw"];

        include_once 'Config.php';
        $conn = mysqli_connect(DB_HOST, DB_USER, DB_PW, DB_NAME); 
        $sql = "SELECT * FROM users WHERE userEmail = '$user_ID' and userPW = '$user_PW'";       // userEmail, userPW check,
        $result = mysqli_query($conn, $sql); 
        if($result){
                $count = mysqli_num_rows($result);                            // If Query exist,
                if($count > 0){
                        $data_row = mysqli_fetch_array($result);
                        echo "$data_row[0], $data_row[1], $data_row[2]";      // return userInfo
                }
                else
                        echo "Fail";
        }else
                echo "Fail";

        mysqli_close($conn);
}
?>
